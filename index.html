Apps Script kód (skopíruj):
const SHEET_NAME = 'Cars';

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) sheet = ss.insertSheet(SHEET_NAME);

  const action = (e.parameter && e.parameter.action) || (e.postData && JSON.parse(e.postData.contents).action);
  const cors = ContentService.createTextOutput;

  if (action === 'read') {
    const data = sheet.getDataRange().getValues();
    const headers = data[0] || [];
    const rows = data.slice(1).map(row => {
      const obj = {};
      headers.forEach((h, i) => { try { obj[h] = JSON.parse(row[i]); } catch { obj[h] = row[i]; } });
      return obj;
    }).filter(r => r.id);
    return ContentService.createTextOutput(JSON.stringify({ok:true,cars:rows})).setMimeType(ContentService.MimeType.JSON);
  }

  if (action === 'write') {
    const body = JSON.parse(e.postData.contents);
    const cars = body.cars || [];
    sheet.clearContents();
    if (cars.length === 0) return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    const keys = Object.keys(cars[0]);
    sheet.appendRow(keys);
    cars.forEach(car => sheet.appendRow(keys.map(k => typeof car[k] === 'object' ? JSON.stringify(car[k]) : car[k])));
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ok:false,error:'Unknown action'})).setMimeType(ContentService.MimeType.JSON);
}
